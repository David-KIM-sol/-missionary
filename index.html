    <script>

        document.addEventListener('DOMContentLoaded', function() {

            const totalCards = 8;

            let currentCard = 1;


            const prevBtn = document.getElementById('prevBtn');

            const nextBtn = document.getElementById('nextBtn');

            const paginationContainer = document.getElementById('pagination');


            // Create pagination dots

            for (let i = 1; i <= totalCards; i++) {

                const dot = document.createElement('button');

                dot.classList.add('w-2', 'h-2', 'rounded-full', 'transition');

                dot.dataset.card = i;

                if (i === 1) {

                    dot.classList.add('bg-white');

                    dot.classList.remove('bg-gray-400');

                } else {

                    dot.classList.add('bg-gray-400');

                    dot.classList.remove('bg-white');

                }

                dot.addEventListener('click', () => showCard(i));

                paginationContainer.appendChild(dot);

            }

            const dots = paginationContainer.querySelectorAll('button');


            function showCard(cardNumber) {

                // Hide current card

                document.getElementById(`card-${currentCard}`).classList.add('opacity-0');

                

                // Show new card

                currentCard = cardNumber;

                document.getElementById(`card-${currentCard}`).classList.remove('opacity-0');


                updateUI();

            }


            function updateUI() {

                 // Update buttons

                prevBtn.disabled = currentCard === 1;

                nextBtn.disabled = currentCard === totalCards;

                

                // Update pagination dots

                dots.forEach(dot => {

                    if (parseInt(dot.dataset.card) === currentCard) {

                        dot.classList.add('bg-white', 'scale-125');

                        dot.classList.remove('bg-gray-400');

                    } else {

                        dot.classList.remove('bg-white', 'scale-125');

                        dot.classList.add('bg-gray-400');

                    }

                });

            }


            prevBtn.addEventListener('click', () => {

                if (currentCard > 1) {

                    showCard(currentCard - 1);

                }

            });


            nextBtn.addEventListener('click', () => {

                if (currentCard < totalCards) {

                    showCard(currentCard + 1);

                }

            });


            // Initialize UI

            updateUI();

        });

    </script>
